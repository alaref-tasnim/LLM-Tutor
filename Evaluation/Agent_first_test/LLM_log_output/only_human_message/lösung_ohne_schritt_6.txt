# String in Liste von Wörtern konvertieren.  
# Eingabe: ein String  
# Ergebnis: Liste mit den Wörtern, die im String enthalten sind.

    satzzeichen = [',', '?', '.', ';', ':', '!']  
    for x in satzzeichen:  
        s = s.replace(x, ' ')  
    return s.split()
