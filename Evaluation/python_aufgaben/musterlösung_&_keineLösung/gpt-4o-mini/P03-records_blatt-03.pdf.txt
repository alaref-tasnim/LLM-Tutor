# Kurzbeschreibung: Implementierung eines Preismodells für Computer.

# Daten- und Recorddefinition für Computer
class Computer:
    def __init__(self, ram_gb: int, hdd_gb: int, cpu: str):
        self.ram_gb = ram_gb
        self.hdd_gb = hdd_gb
        self.cpu = cpu

# Funktionskopf
def preisberechnung(computer: Computer) -> float:
    # Tests mittels check (hier als Beispiel)
    assert isinstance(computer, Computer), "Eingabe muss ein Computer sein."
    
    # Passende Schablonen
    preis = 0
    preis += max(30, computer.ram_gb * 5)
    preis += ( (computer.hdd_gb + 499) // 500) * 20  # Aufrunden auf das nächste 500 GB
    preis += 90  # CPU Grundpreis
    if 'Intel' in computer.cpu:

    preis += 100  # Pauschale für Komponenten und Montage
    
    return preis
