def pizza_preis(groesse, zutaten):
    grundpreise = {'normal': 2.0, 'gross': 2.50}
    zutaten_preise = {'oliven': 0.50, 'pilze': 0.40, 'salami': 0.45, 'kaese': 0.55}

    preis = grundpreise[groesse]
    if groesse == 'gross':
        preis *= 1.1  # 10% Aufpreis für große Pizzen
    
    for zutat in zutaten:
        preis += zutaten_preise[zutat]
    
    return preis

def gericht_preis(typ, **kwargs):
    if typ == 'pizza':
        return pizza_preis(kwargs['groesse'], kwargs['zutaten'])
    elif typ == 'bier':
        preis_pro_100ml = {'pils': 0.80, 'weizen': 0.85}
        return preis_pro_100ml[kwargs['sorte']] * kwargs['menge'] / 100
    elif typ == 'nudeln':
        return kwargs['preis_pro_100g'] * kwargs['menge'] / 100
    else:
        return 0
