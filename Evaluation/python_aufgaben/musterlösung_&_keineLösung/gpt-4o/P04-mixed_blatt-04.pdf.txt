```python
# Diese Funktion berechnet den Preis einer Pizza basierend auf Größe und Zutaten.
# Parameter: größe (str: 'normal' oder 'groß'), zutaten (set: mögliche Werte 'Oliven', 'Pilze', 'Salami', 'Käse')
# Ergebnis: der berechnete Preis der Pizza (float)

def berechne_pizza_preis(groesse: str, zutaten: set) -> float:
    # Festlegen der Grundpreise
    grundpreise = {'normal': 2.0, 'groß': 2.5}
    zutaten_preise = {'Oliven': 0.5, 'Pilze': 0.4, 'Salami': 0.45, 'Käse': 0.55}

    # Grundpreis basierend auf der Größe
    preis = grundpreise[groesse]

    # Hinzufügen der Preise der Zutaten
    for zutat in zutaten:
        preis += zutaten_preise[zutat]

    # Aufpreis von 10% für große Pizzen
    if groesse == 'groß':
        preis *= 1.1

    return preis

# Tests zum Überprüfen der Pizza-Preis-Funktion
check(berechne_pizza_preis('normal', {'Oliven', 'Käse'}), 3.05)
check(berechne_pizza_preis('groß', {'Pilze', 'Salami'}), 3.47)

# Diese Funktion berechnet den Preis eines Gerichts oder Getränks.
# Parameter: Art des Artikels (str: 'Pizza', 'Bier', 'Nudel'), details (dict)
# Ergebnis: der berechnete Preis (float)

def berechne_gesamt_preis(art: str, details: dict) -> float:
    if art == 'Pizza':
        return berechne_pizza_preis(details['groesse'], details['zutaten'])
    elif art == 'Bier':
        # Bierpreisberechnung basierend auf Sorte und Menge
        sorte_preise = {'Pils': 0.8, 'Weizen': 0.85}
        menge = details['menge']  # in ml
        preis_pro_100ml = sorte_preise[details['sorte']]
        return (menge / 100) * preis_pro_100ml
    elif art == 'Nudel':
        # Nudelpreisberechnung basierend auf Preis pro 100g und Bestellmenge in g
        preis_pro_100g = details['preis_pro_100g']
        menge = details['menge']  # in g
        return (menge / 100) * preis_pro_100g

# Tests zum Überprüfen der Gesamtpreis-Funktion

check(berechne_gesamt_preis('Bier', {'sorte': 'Weizen', 'menge': 500}), 4.25)
check(berechne_gesamt_preis('Nudel', {'preis_pro_100g': 1.5, 'menge': 200}), 3.0)
```
