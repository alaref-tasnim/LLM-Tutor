# Diese Funktion entfernt Elemente aus einer Liste, die von einem Slice-Ausdruck bestimmt werden.
# Parameter: l - eine zu verändernde Liste (List), start - Anfangsindex des Slice (int), end - Endindex des Slice (int), step - Schrittweite des Slice (int)
# Ergebnis: Die Liste l wird direkt modifiziert, keine Rückgabe (None)

def removeSlice(l: list, start: int, end: int, step: int) -> None:
    # Identifiziere die zu entfernenden Indizes
    indices_to_remove = set(range(start, end, step))
    # Entferne die Elemente an den spezifischen Indizes
    l[:] = [item for idx, item in enumerate(l) if idx not in indices_to_remove]
